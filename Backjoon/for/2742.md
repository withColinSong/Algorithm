# 2. 코드
```
public class Main {
    public static void main(String[] args) {

       BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        try {

            int n = Integer.parseInt(br.readLine());

            for(int i=0; i<n; i++) {
                System.out.println(n-i);
            }

        } catch (IOException e) {
            e.printStackTrace();
        }

    }
}
```
